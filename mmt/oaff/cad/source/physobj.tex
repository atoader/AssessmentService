\begin{module}[id=physobj]

\begin{omtext}
  The following definitions are circular, they describe very fundamental concepts that
  are very hard to properly define. We take a relatively naive approach here, where
  space-time is Euclidean and mater is made up by elementary particles.
\end{omtext}

\begin{definition}
  \adefi{Matter}{matter} is generally anything that has mass and \trefi{extent}. Matter is
  said to exist in three principal \defi{state}s depending on the bonding of the particles
  that make up the matter:
  \begin{itemize}
  \item \defi{solid}: the particles are tightly bound by forces that prevent movement
    against each other and make the matter rigit. 
  \item \defi{liquid}: the particles are bound less tightly by forces, so that the
    particles can move against each other but are still strong enough to full separation. 
  \item \defi{gaseous}: the particles are vastly separated from each other.
  \end{itemize}
\end{definition}

\begin{definition}
  \atrefi{Gases}{gas} and \trefi{liquid}s are together called \defi{fluid}s
\end{definition}

\begin{definition}
  A (chemical) \defi{substance} is a form of \trefi{matter} that has constant chemical
  composition and characteristic properties.  It cannot be separated into components by
  physical separation methods, i.e. without breaking chemical bonds.
\end{definition}

\begin{definition}
  A \defii{physical}{object} is a collection of \defi{particle}s (pieces of
  \trefi{matter}) taken to be one; an entity that can be perceived by the senses and acted
  upon. Physical objects are characterized by their extent in time and space and the
  matter they consist of.
\end{definition}

\begin{definition}
  The \defi{extent} of a physical object is the space it takes up seen over time. This is
  taken to be a function from time into subsets of three-dimensional space.
\end{definition}

\begin{omtext}
  Often the extent function is constant up to translations and rotations (moving the
  object around in space).
\end{omtext}

\begin{definition}
  The \defi{geometry} of a physical object (at a given point in time $t$) is the
  equivalence class of its extent at $t$ under all euclidean translations and rotations.
\end{definition}

\end{module}
