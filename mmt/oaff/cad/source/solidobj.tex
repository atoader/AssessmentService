\begin{module}[id=solidobj]
\importmodule[load=\FCAD{flange/cds/physobj}]{physobj}
\begin{definition}
  A \defii{solid}{object} is a \trefii[physobj]{physical}{object} that is
  \trefi[physobj]{solid}. 

  As the \trefi[physobj]{extent} of solid objects is constant over time (unless subjected
  to external forces), we can speak of the \defi{shape} of a solid object as its current
  extent.
\end{definition}

\symdef{solidobjects}{\text{SObj}}
\begin{definition}
  The \defi{type} of a solid object is usually characterized by classifying objects by
  their shape and substance.
\end{definition}
\end{module}
