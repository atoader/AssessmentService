\begin{module}[id=color]
\importmodule[load=\FCAD{flange/cds/physobj}]{physobj}
\begin{definition}
  The \defi{color} of a \trefii[physobj]{physical}{object} is defined by the spectrum of
  lights it emits or reflects. It is determined by the object's substance, its temperature
  and the spectrum of the incoming light.
\end{definition}
\end{module}
