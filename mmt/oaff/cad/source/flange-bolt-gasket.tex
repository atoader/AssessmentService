\begin{module}[id=flange-bolt-gasket]
\importmodule[load=\FCAD{flange/cds/nutbolt}]{nutbolt}
\begin{definition}
  A \defi{flange} is an external or internal edge or rim on an object, e.g. a flange on a
  pipe as a connector.
\end{definition}

\begin{definition}
  A \defii{blind}{flange} is a plate for covering or closing the end of a pipe, with a
  diameter larger than the pipe's.
\end{definition}

\begin{definition}
  A \defi{gasket} is a mechanical seal which fills the space between two or more
  mating surfaces, generally to prevent leakage from or into the joined objects while
  under compression.
\end{definition}

\begin{definition}
  A \defii{flange}{gasket} is a type of \trefi{gasket} made to fit between two sections of
  pipe that are flared to provide higher surface area, or between a pipe section and a
  \trefii{blind}{flange}.
\end{definition}

\begin{definition}
  A \defi{flange-bolt-gasket} system is an assembly that covers or closes the flared end
  of a pipe, consisting of a \trefii{blind}{flange}, a \trefi{gasket} that seals the space
  between the flange and the pipe, and a number of \atrefi[nutbolt]{bolts}{bolt} that hold
  together the flange, gasket and pipe end.
\end{definition}
\end{module}
