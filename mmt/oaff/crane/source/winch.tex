\begin{module}[id=winch]
\importmodule[load=\FCAD{crane/cds/cable}]{cable}

\begin{definition}
  A \defi{winch} is a mechanical device that is used to pull in (wind up) or let out (wind
  out) or otherwise adjust the tension of a \trefi[cable]{cable}. A winch consists of a
  \defi{spool} (also called \defii[spool]{winch}{drum}) and a \defi{crank} or
  \trefi{motor}.
\end{definition}
\end{module}
