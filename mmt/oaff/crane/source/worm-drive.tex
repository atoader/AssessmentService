\begin{module}[id=worm-drive]
\importmodule[load=\FCAD{crane/cds/transmission}]{transmission}
\importmodule[load=\FCAD{crane/cds/gear}]{gear}
  \begin{definition}
    A \defii{worm}{drive} is a \trefii[transmission]{transmission} in which a \defi{worm}
    (which is a \trefi[gear]{gear} in the form of a screw) meshes with a
    \defii{worm}{gear} (also called a \defii[worm-gear]{worm}{wheel}).
  \end{definition}
\end{module}
