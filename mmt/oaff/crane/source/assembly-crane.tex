\begin{module}[id=assembly-crane]
  \importmodule[load=\FCAD{crane/cds/crane}]{crane}
  \begin{definition}
    An \defii{assembly}{crane} is a small type of \trefi[crane]{crane} used for lifting
    heavy building elements and machine components in workshops. It usually consists of a
    \defii{base}{frame} supports all components of the \trefi[crane]{assembly}{crane} and
    transfers the forces induced by the load to be floor. The base frame is often realized
    as a two-pronged fork structure with two \defi{leg}s.

    The vertical \defi{mast} supports the \defi{jib}, a horizontal beam that carries the
    \trefi[sheave]{sheave}s for the \trefi[cable]{cable} which carries the
    \trefii[hoist]{lifting}{hook}.
\end{definition}

\end{module}
