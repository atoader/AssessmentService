\begin{module}[id=gear]
\begin{definition}
  A \defi{gear} or \defi[gear]{cogwheel} is a rotating machine part having cut
  \adefi{teeth}{tooth}, or \adefi[tooth]{cogs}, which mesh with another toothed part in
  order to transmit torque, in most cases with teeth on the one gear of identical shape,
  and often also with that shape (or at least width) on the other gear.
\end{definition}
\end{module}
