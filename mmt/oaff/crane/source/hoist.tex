\begin{module}[id=hoist]
\importmodule[load=\FCAD{crane/cds/cable}]{cable}
  \begin{definition}
    A \defi{hoist} is a device used for lifting or lowering a load by means of a drum or
    lift-wheel around which \trefi[rope]{rope}, \trefi[cable]{cable}, or
    \trefi[chain]{chain} wraps. It may be manually operated, electrically or pneumatically
    driven and may use chain, fiber or wire rope as its lifting medium. The load is
    attached to the hoist by means of a \trefii{lifting}{hook}.
  \end{definition}

  \begin{definition}
    A \defii{lifting}{hook} is a device for grabbing and lifting loads by means of a
    \trefi{hoist}. A lifting hook is usually equipped with a safety latch to prevent the
    disengagement of the lifting wire rope sling, chain or rope to which the load is
    attached. A lifting hook may have one or more built-in pulleys to amplify the lifting
    force.
  \end{definition}
\end{module}
