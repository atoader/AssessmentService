\begin{module}[id=sheave]
\importmodule[load=\FCAD{crane/cds/cable}]{cable}
\begin{definition}
  A \defi{sheave} is a wheel or roller with a groove along its edge for holding a belt,
  rope, or \trefi[cable]{cable}. When hung between two supports and equipped with a belt,
  rope or \trefi[cable]{cable}, one or more sheaves make up a \defi{pulley}.
\end{definition}
\end{module}
