\begin{module}[id=crane]
\importmodule[load=\FCAD{crane/cds/sheave}]{sheave}
\importmodule[load=\FCAD{crane/cds/winch}]{winch}
\importmodule[load=\FCAD{crane/cds/hoist}]{hoist}
\begin{definition}
  A \defi{crane} is a type of machine, generally equipped with a \trefi[hoist]{hoist} or
  \trefi[winch]{winch}, \trefi[cable]{cable}s or \trefi[rope]{rope}s, and
  \trefi[sheave]{sheave}s, that can be used both to lift and lower materials and to move
  them horizontally. It is mainly used for lifting heavy things and transporting them to
  other places.
\end{definition}

\end{module}
